{{/* This partial expects to receive a map with the Styles and Settings keys. The Styles key should be an array of maps or strings representing SCSS style modules to load for adding fonts to the site. The Settings key should be a map of options for defining selectors without having to write them in SCSS. */}} {{- $Params := . -}} {{- $FontStyles := $Params.styles -}} {{- $FontSettings := $Params.settings -}} {{- $FontUrlPattern := `^https?:\/\/` -}} {{/* Initialize the slice of strings to render */}} {{- $renderStrings := slice -}} {{/* Build the lists of weighted and unweighted font styles. Weighted font styles are added first, from lowest to highest weight. Unweighted styles are added afterward in alphabetical order. */}} {{- $weightedFontStyles := slice -}} {{- $unweightedFontStyles := slice -}} {{- range $Style := $FontStyles -}} {{- $IsMap := reflect.IsMap $Style -}} {{- if (and $IsMap (isset $Style "weight")) -}} {{- $weightedFontStyles = $weightedFontStyles | append $Style -}} {{- else if $IsMap -}} {{- $unweightedFontStyles = $unweightedFontStyles | append $Style.import_name -}} {{- else -}} {{- $unweightedFontStyles = $unweightedFontStyles | append $Style -}} {{- end -}} {{- end -}} {{/* Add import statements for weighted font styles. If a style has an explicitly nil weight, it's functionally disabled, so ignore those. */}} {{- range $Font := sort (where $weightedFontStyles "weight" "ne" nil) "weight" -}} {{- $importLine := printf `@import "fonts/%s";` $Font.import_name -}} {{- if findRE $FontUrlPattern $Font.import_name -}} {{- $importLine = printf `@import url('%s');` $Font.import_name -}} {{- end -}} {{- $renderStrings = $renderStrings | append $importLine -}} {{ end }} {{/* Add import statements for unweighted font styles. */}} {{- range $Font := sort $unweightedFontStyles -}} {{- $importLine := printf `@import "fonts/%s";` $Font -}} {{- if findRE $FontUrlPattern $Font -}} {{- $importLine = printf `@import url('%s');` $Font -}} {{- end -}} {{- $renderStrings = $renderStrings | append $importLine -}} {{- end -}} {{/* Add selector declarations for every defined setting */}} {{- range $Name, $Settings := $FontSettings -}} {{- $Selector := default (lower $Name) $Settings.selector -}} {{- $selectorString := slice (printf "%s {" $Selector) -}} {{/* Add the family declaration, if any, for this selector */}} {{- with $Settings.family -}} {{- $selectorString = $selectorString | append (printf "font-family: %s;" $Settings.family) -}} {{- end -}} {{/* Add every declared mixin for this selector */}} {{- range $MixIn := $Settings.mixins -}} {{- $selectorString = $selectorString | append (printf "@include %s;" $MixIn) -}} {{- end -}} {{/* Close the selector and delimit it with appropriate indentation for rendering */}} {{- $selectorString = $selectorString | append "}" -}} {{- $renderStrings = $renderStrings | append (delimit $selectorString "\n " "\n") -}} {{/* Temporary special handling for body to conform with Shoelace */}} {{- if and (eq $Name "body") (isset $Settings "family") -}} {{- $sansFontOverride := printf `--sl-font-sans: %s;` $Settings.family -}} {{- $sansFontOverride = printf ":root {\n %s\n}" $sansFontOverride -}} {{- $renderStrings = $renderStrings | append $sansFontOverride -}} {{- end -}} {{/* Temporary special handling for code to conform with Shoelace. */}} {{- if and (eq $Name "code") (isset $Settings "family") -}} {{- $monoFontOverride := printf `--sl-font-mono: %s;` $Settings.family -}} {{- $monoFontOverride = printf ":root {\n %s\n}" $monoFontOverride -}} {{- $renderStrings = $renderStrings | append $monoFontOverride -}} {{- end -}} {{ end }} {{/* Return the SCSS to render */}} {{- $renderStrings = delimit $renderStrings "\n\n" | safeCSS -}} {{- return $renderStrings -}}
