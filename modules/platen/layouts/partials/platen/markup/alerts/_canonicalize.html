You can specify values with dotpath notation as needed, but they should always be relative to the markup's configuration key. So far the config setting 'platen.markup.foo.bar.baz', specify the key as "foo.bar.baz" */}} {{- $ConfigOptions := slice "custom" "classes" "variant" "use_default_icons" -}} {{/* Define the list of default values for the options that aren't pulled from the site configuration. If you need to define defaults, add them to the dictionary using the option's key and default value. For example, to default the legacy option to false, add this to the $NonConfigDefaults dictionary: "legacy" false */}} {{- $NonConfigDefaults := dict "closable" false "open" true "variant" "default" -}} {{/* Initialize the return value for this partial */}} {{- $canonical := dict -}} {{/* Canonicalize options with config defaults, using the value from the config unless specified by the markup options (already merged with preset). It loops over the list of config options to see if they're set. Then it uses those values as the default for the matching value from the merged options, ignoring the value from config if it was set in the markup's attribute, options, or preset properties. You can check the name of the key to do specific munging if the values in the configuration don't match the value type for the options. Here we left the handling for classes, where the configuration specifies `classes` as an array but markup always uses `class` due to the built-in handling of class attributes by Hugo. It space-delimits the values from config and uses them as a default if `class` isn't defined in the options. You can do similar handling for other keys by adding an 'else if' statement before the end of the conditional, like: {{- else if (eq "use_legacy" $OptionKey) -}} {{- $key = "legacy" -}} {{- $value = $options.legacy | default $value -}} {{- else -}} Which would check for a key called 'use_legacy' and use its value for the 'legacy' key in the options. */}} {{- range $OptionKey := $ConfigOptions -}} {{- if isset $Config $OptionKey -}} {{- $key := $OptionKey -}} {{- $value := index $Config $OptionKey -}} {{- if eq "classes" $OptionKey -}} {{- $key = "class" -}} {{- $value = $options.class | default (delimit $value " ") -}} {{- else -}} {{- $value = index $options $OptionKey | default $value -}} {{- end -}} {{/* Merge the munged value in */}} {{- $options = merge $options (dict $key $value) -}} {{- end -}} {{- end -}} {{/* Loop over the default values that aren't defined by the configuration and set if needed. */}} {{- range $Key, $Value := $NonConfigDefaults -}} {{- if not (isset $options $Key) -}} {{- $options = merge $options (dict $Key $Value) -}} {{- end -}} {{- end -}} {{/* Validate and munge the definition as necessary. Alerts always require a definition, but it might be passed from a preset instead of in the codeblock itself. */}} {{- with $definition -}} {{/* Nothing to do, definition passed directly */}} {{- else -}} {{- with $options.definition -}} {{/* definition passed through a preset */}} {{- $definition = $options.definition -}} {{- else -}} {{- errorf "Invalid alert markup at %s: missing a definition to render as the alert's content." $Position -}} {{- end -}} {{- end -}} {{/* This markup always requires shoelace. Mark the page so it gets loaded. */}} {{- $Page.Store.Set "_hasShoelace" true -}} {{/* Retrieve and trim the actual content definition for the alert's codeblock and pretty-render it. This trims the leading text from the definition, as well as extra newlines before and after the content. Then it renders the Markdown as a block (if it's only one line, it gets rendered in a

tag), indented 4 spaces. Finally, add the rendered content to the canonical map as `Content`. */}} {{- $TrimParams := dict "Text" $definition "TrimEOL" true -}} {{- $content := partial $getLeadTrimmedText $TrimParams -}} {{- $RenderParams := dict "Page" $Page "Content" $content "IndentCount" 4 "AsBlock" true -}} {{- $content = partial $getPrettyRender $RenderParams -}} {{- $canonical = merge $canonical (dict "Content" $content) -}} {{/* If the markup specified a header, render it and add to canonical map. We render it as inline Markdown, because it's going inside a span. While the header *can* be a block element, we shouldn't assume that it is. Most of the time, it won't be. */}} {{- with $options.header -}} {{- $RenderParams := dict "Page" $Page "Content" $options.header "IndentCount" 6 "AsBlock" false -}} {{- $Header := partial $getPrettyRender $RenderParams -}} {{- $canonical = merge $canonical (dict "Header" $Header) -}} {{- end -}} {{/* This canonicalizes a list of classes, merging the markup's default(s) with those specified in the options. We include this in the canonicalized map even though it's not used by the default template, because it's useful for the custom template. */}} {{- $DefaultClasses := slice "platen-alert" -}} {{- $MergingParams := dict "Default" $DefaultClasses "Additional" $options.class -}} {{- $Classes := partial $getMergedClasses $MergingParams -}} {{- $canonical = merge $canonical (dict "Classes" $Classes) -}} {{/* This builds a list of attributes and turns them into an insertable list for your template. Make sure you actually handle the attributes that need to be inserted, this is just an example. You can also pass the attribute values directly into the canonical map to insert in the template. Note that we're always appending the full attribute declaration, including name and value, as a string to the list of attributes. */}} {{- $attributes := slice (printf `variant="%s"` $options.variant) -}} {{/* Handle the alert's classes. */}} {{- with $Classes -}} {{- $attributes = $attributes | append (printf `class="%s"` $Classes) -}} {{- end -}} {{/* Handle the alert's id. */}} {{- with $options.id -}} {{- $attributes = $attributes | append (printf `id="%s"` $options.id) -}} {{- end -}} {{/* Handle whether the alert should be open. If this is false, the alert isn't visible. Setting this to false is less useful for general alerts, but useful when building components or adding an alert that can be triggered with a button. */}} {{- if $options.open -}} {{- $attributes = $attributes | append `open` -}} {{- end -}} {{/* Handle the closable option for the alert, allowing a user to choose to hide it from view by clicking a button. It returns if the page is refreshed. */}} {{- if $options.closable -}} {{- $attributes = $attributes | append `closable` -}} {{- end -}} {{/* Handle the duration, which makes the alert close after it expires. This is less useful for general alerts, but useful when building components or adding an alert that can be triggered with a button. */}} {{- with $options.duration -}} {{- $count := 0 -}} {{- $unit := "s" -}} {{/* Parse the passed duration with regex to find the count and unit: ^ This anchors to the beginning of the string. It ensures that there's no text before our pattern, so `bagel54` won't match. (\d+) The first group looks for one or more numerals. This is the count for the duration. ([a-zA-Z]+)? The second group looks for an optional suffix of one or more letters, case insensitive. This is the unit for the duration. $ This anchors to the end of the string. It ensures that there's no text after our pattern, so `54ms-foo` won't match. */}} {{- $Pattern := `^(\d+)([a-zA-Z]+)?$` -}} {{- $Results := findRESubmatch $Pattern (print $options.duration) -}} {{- with $Results -}} {{/* If there was a match, Hugo returns an array of matches as the result. There can only actually be one valid match, so index into the first array to get the matches. Index 0 is the full string from the match, which is the entire duration value specified in the options. We ignore it. Index 1 is the numerals for the count. We cast it to an integer so we can multiply it, if the duration is in seconds. Index 2 is the optional unit specifier. If it wasn't specified in the duration value, we default to `s`. We always lower case it so we can compare it to the list of valid values - Hugo is case-sensitive. */}} {{- $Matches := index $Results 0 -}} {{- $count = index $Matches 1 | int -}} {{- $unit = index $Matches 2 | default "s" | lower -}} {{/* If the value specified an invalid duration, like `1m`, warn the user and default to seconds. */}} {{- if not (in (slice "s" "ms") $unit) -}} {{- $Message := delimit ( slice (printf "Specified invalid unit '%s' for alert duration at %s." $unit $Position) "Valid units are 's' (seconds) and 'ms' (milliseconds);" (printf "defaulting to seconds for this alert, making the duration %v seconds." $count) ) " " | print -}} {{- warnf $Message -}} {{- $unit = "s" -}} {{- end -}} {{/* For seconds, multiply the value by 1000. The sl-alert element expects the duration to be specified in milliseconds. */}} {{- if eq $unit "s" -}} {{- $count = mul $count 1000 -}} {{- end -}} {{/* Add the duration attribute to the list now that it's validated and munged. */}} {{- $attributes = $attributes | append (printf `duration="%v"` $count) -}} {{- else -}} {{/* Hitting this conditional indicates the duration didn't match the regex. */}} {{- $Message := delimit ( slice (printf "Specified invalid value '%v' for alert duration at %s." $options.duration $Position) "Valid durations are a number with an optional unit, either 's' (seconds) or 'ms' (milliseconds)." "For example 5s for five seconds or 2500ms for two-and-a-half seconds." "Ignoring the duration for this alert." ) " " | print -}} {{- warnf $Message -}} {{- end -}} {{- end -}} {{/* With the attribute list defined, convert into a string for the template and canonicalize the attributes. The getSafeAttributes partial is used to ensure the attributes are split across lines with the correct indent as needed; this makes the printed HTML prettier in raw form and in the docs examples. */}} {{- $AttributeParams := dict "Attributes" $attributes "IndentCount" 10 -}} {{- $attributes = partial $getSafeAttributes $AttributeParams -}} {{- $canonical = merge $canonical (dict "Attributes" $attributes) -}} {{/* Handle the icon attributes - if an icon is defined, it should at least get the name attribute. If this dict stays empty, no icon gets added. */}} {{- $iconAttributeMap := dict -}} {{/* The icon can be specified as either a string or a map for more control. If it's a string, use the value for the `name` key in the attribute map. */}} {{- with $options.icon -}} {{- if reflect.IsMap $options.icon -}} {{- $iconAttributeMap = merge $iconAttributeMap $options.icon -}} {{- else -}} {{- $iconAttributeMap = merge $iconAttributeMap (dict "name" $options.icon) -}} {{- end -}} {{- end -}} {{/* If it should use default icons, try to use that unless an icon was specified in the markup or a preset. */}} {{- if $options.use_default_icons -}} {{- $ConfigIcon := index $Config.icons $options.variant -}} {{- with $iconAttributeMap.name -}} {{/* Nothing to do, already set. */}} {{- else -}} {{- $iconAttributeMap = merge $iconAttributeMap (dict "name" $ConfigIcon) -}} {{- end -}} {{- end -}} {{/* If the icon attributes have been set, build the attribute slice up. If it wasn't, error only if the markup expected to use the default icons and the icon for this alert's variant couldn't be found. */}} {{- with $iconAttributeMap -}} {{- $iconAttributes := slice -}} {{/* Loop over the defined keys, turning them into attribute declarations. We always print the value before insertion, because HTML attributes expect the values as strings. */}} {{- range $Key, $Value := $iconAttributeMap -}} {{- $IconAttribute := printf `%s="%s"` $Key (print $Value) -}} {{- $iconAttributes = $iconAttributes | append $IconAttribute -}} {{- end -}} {{/* Convert the slice into an insertable set of attributes. We need to make sure the attributes are returned one-per-line with an eleven-space indent before the attribute. This ensures they pretty-print in the raw HTML, which is displayed in the docs examples. */}} {{- $IconParams := dict "Attributes" $iconAttributes "IndentCount" 11 "MultiLine" true -}} {{- $iconAttributes = partial $getSafeAttributes $IconParams -}} {{- $canonical = merge $canonical (dict "IconAttributes" $iconAttributes) -}} {{- else -}} {{- if $options.use_default_icons -}} {{- $Message := delimit ( slice (printf "Tried to use default icon in alert markup at %s for variant '%s'," $Position $options.variant) "but none was defined. Is there an entry in your site config for" (printf "'params.platen.markup.alerts.icons.%s'?" $options.variant) "You can add it in 'data/_params/platen/markup.yaml' too." ) " " | print -}} {{- warnf $Message -}} {{- end -}} {{- end -}} {{/* Handle the custom template option. */}} {{- if $options.custom -}} {{/* Define context to pass; it's best to pass as much info as you can for a custom template, since you don't know what they'll find useful.
